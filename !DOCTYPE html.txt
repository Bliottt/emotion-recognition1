    <script>
        let video = document.getElementById('video');
        let stream = null;
        let isRunning = false;

        const emotions = {
            happy: 'üòä',
            sad: 'üò¢',
            angry: 'üò†',
            surprised: 'üòÆ',
            neutral: 'üòê'
        };

        async function startCamera() {
            try {
                stream = await navigator.mediaDevices.getUserMedia({
                    video: { facingMode: 'user' }
                });
                video.srcObject = stream;
                document.getElementById('startBtn').style.display = 'none';
                document.getElementById('stopBtn').style.display = 'inline-block';
                isRunning = true;
                detectEmotions();
            } catch (err) {
                console.error('Error:', err);
                alert('Êó†Ê≥ïËÆøÈóÆÁõ∏Êú∫ÔºåËØ∑Á°Æ‰øùÂ∑≤ÊéàÊùÉ„ÄÇ');
            }
        }

        function stopCamera() {
            if (stream) {
                stream.getTracks().forEach(track => track.stop());
                video.srcObject = null;
                document.getElementById('startBtn').style.display = 'inline-block';
                document.getElementById('stopBtn').style.display = 'none';
                isRunning = false;
            }
        }

        function detectEmotions() {
            if (!isRunning) return;
            const emotionList = Object.values(emotions);
            const randomEmoji = emotionList[Math.floor(Math.random() * emotionList.length)];
            document.getElementById('emoji').textContent = randomEmoji;
            setTimeout(detectEmotions, 500);
        }

        window.onbeforeunload = stopCamera;
